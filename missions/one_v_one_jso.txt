one_v_one_mission_jap_1_jso = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		NOT = { map_setup = map_setup_random }
		OR = { tag = JSO tag = JSS tag = JSM tag = JST tag = JNU }
	}
	has_country_shield = yes

	one_v_one_mission_jap_nation_select_1_jso = {
		icon = one_v_one_mission_flag_jss
		required_missions = { }
		trigger = {
			if = {
				limit = { OR = { has_country_flag = jso_chose_nation_jss_flag has_country_flag = jso_chose_nation_jsm_flag has_country_flag = jso_chose_nation_jst_flag has_country_flag = jso_chose_nation_jnu_flag } }
				custom_trigger_tooltip = {
					tooltip = one_v_one_already_chose_nation_flag_tt
					always = no
				}
			}
			else = { OR = { tag = JSO tag = JSS } }
			custom_trigger_tooltip = {
				tooltip = one_v_one_nation_choosing_flag_tt
				has_global_flag = jno_nation_choosing_flag
				has_global_flag = jso_nation_choosing_flag 
			}
			if = {
				limit = { has_global_flag = jno_ban_jss_flag }
				custom_trigger_tooltip = {
					tooltip = one_v_one_nation_banned_flag_tt
					always = no
				}
			}
		}
		effect = {
			set_country_flag = jso_chose_nation_jss_flag
			set_global_flag = jso_chose_nation_flag
			custom_tooltip = one_v_one_chose_nation_flag_tt
			tooltip = { one_v_one_jso_mission_select_jss_tag = yes }
			hidden_effect = {
				if = {
					limit = { has_global_flag = jno_chose_nation_flag }
					one_v_one_jso_mission_select_jss_tag = yes
					JNO = {
						if = {
							limit = { has_country_flag = jno_chose_nation_jnd_flag }
							one_v_one_jno_mission_select_jnd_tag = yes
						}
						if = {
							limit = { has_country_flag = jno_chose_nation_jnk_flag }
							one_v_one_jno_mission_select_jnk_tag = yes
						}
						if = {
							limit = { has_country_flag = jno_chose_nation_jnt_flag }
							one_v_one_jno_mission_select_jnt_tag = yes
						}
						if = {
							limit = { has_country_flag = jno_chose_nation_jsi_flag }
							one_v_one_jso_mission_select_jsi_tag = yes
						}
					}
				}
			}
		}
	}
}
