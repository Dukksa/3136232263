one_v_one_mission_gtw_1_gpl = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		NOT = { map_setup = map_setup_random }
		tag = GPL
	}
	has_country_shield = yes

	one_v_one_mission_gtw_nation_select_1_gpl = {
		icon = mission_conquer_5_states
		required_missions = { }
		trigger = {
			if = {
				limit = { OR = { has_country_flag = gpl_chose_nation_sobieski_flag has_country_flag = gpl_chose_nation_jablonowski_flag has_country_flag = gpl_chose_nation_sapieha_flag } }
				custom_trigger_tooltip = {
					tooltip = one_v_one_already_chose_nation_flag_tt
					always = no
				}
			}
			else = { tag = GPL }
			custom_trigger_tooltip = {
				tooltip = one_v_one_nation_choosing_flag_tt
				has_global_flag = gto_nation_choosing_flag
				has_global_flag = gec_nation_choosing_flag
				has_global_flag = geb_nation_choosing_flag
				has_global_flag = gea_nation_choosing_flag
				has_global_flag = gfr_nation_choosing_flag
				has_global_flag = gau_nation_choosing_flag
				has_global_flag = gpl_nation_choosing_flag
				has_global_flag = git_nation_choosing_flag
				has_global_flag = gsp_nation_choosing_flag
				has_global_flag = gru_nation_choosing_flag
			}
			if = {
				limit = { NOT = { has_global_flag = gtw_ban_chosen_flag } }
				hidden_trigger = {
					if = {
						limit = { has_global_flag = geb_ban_sobieski_flag }
						custom_trigger_tooltip = {
							tooltip = one_v_one_nation_banned_flag_tt
							always = no
						}
					}
				}
			}
			else = {
				if = {
					limit = { has_global_flag = geb_ban_sobieski_flag }
					custom_trigger_tooltip = {
						tooltip = one_v_one_nation_banned_flag_tt
						always = no
					}
				}
			}
		}
		effect = {
			set_country_flag = gpl_chose_nation_sobieski_flag
			set_global_flag = gpl_chose_nation_flag
			custom_tooltip = one_v_one_chose_nation_flag_tt
			tooltip = { one_v_one_gpl_mission_select_sobieski_tag = yes }
			hidden_effect = {
				if = {
					limit = { has_global_flag = gto_chose_nation_flag has_global_flag = gec_chose_nation_flag has_global_flag = geb_chose_nation_flag has_global_flag = gea_chose_nation_flag has_global_flag = gfr_chose_nation_flag has_global_flag = gau_chose_nation_flag has_global_flag = git_chose_nation_flag has_global_flag = gsp_chose_nation_flag has_global_flag = gru_chose_nation_flag }
					one_v_one_gpl_mission_select_sobieski_tag = yes
					one_v_one_mission_effect_gtw = yes
				}
			}
		}
	}
}
