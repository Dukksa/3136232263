one_v_one_mission_nsi_1_nhy = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		NOT = { map_setup = map_setup_random }
		tag = NHY
	}
	has_country_shield = yes

	one_v_one_mission_nsi_nation_select_1_nhy = {
		icon = mission_janissaries
		required_missions = { }
		trigger = {
			if = {
				limit = { OR = { has_country_flag = nhy_chose_nation_asaf_flag has_country_flag = nhy_chose_nation_nasir_flag has_country_flag = nhy_chose_nation_nizam_flag } }
				custom_trigger_tooltip = {
					tooltip = one_v_one_already_chose_nation_flag_tt
					always = no
				}
			}
			else = { tag = NHY }
			custom_trigger_tooltip = {
				tooltip = one_v_one_nation_choosing_flag_tt
				has_global_flag = nir_nation_choosing_flag
				has_global_flag = nka_nation_choosing_flag
				has_global_flag = nma_nation_choosing_flag
				has_global_flag = nra_nation_choosing_flag
				has_global_flag = nbc_nation_choosing_flag
				has_global_flag = nmu_nation_choosing_flag
				has_global_flag = nbu_nation_choosing_flag
				has_global_flag = nhy_nation_choosing_flag
				has_global_flag = nmy_nation_choosing_flag
				has_global_flag = nfc_nation_choosing_flag
			}
			if = {
				limit = { NOT = { has_global_flag = nsi_ban_chosen_flag } }
				hidden_trigger = {
					if = {
						limit = { has_global_flag = nma_ban_asaf_flag }
						custom_trigger_tooltip = {
							tooltip = one_v_one_nation_banned_flag_tt
							always = no
						}
					}
				}
			}
			else = {
				if = {
					limit = { has_global_flag = nma_ban_asaf_flag }
					custom_trigger_tooltip = {
						tooltip = one_v_one_nation_banned_flag_tt
						always = no
					}
				}
			}
		}
		effect = {
			set_country_flag = nhy_chose_nation_asaf_flag
			set_global_flag = nhy_chose_nation_flag
			custom_tooltip = one_v_one_chose_nation_flag_tt
			tooltip = { one_v_one_nhy_mission_select_asaf_tag = yes }
			hidden_effect = {
				if = {
					limit = { has_global_flag = nir_chose_nation_flag has_global_flag = nka_chose_nation_flag has_global_flag = nma_chose_nation_flag has_global_flag = nra_chose_nation_flag has_global_flag = nbc_chose_nation_flag has_global_flag = nmu_chose_nation_flag has_global_flag = nbu_chose_nation_flag has_global_flag = nmy_chose_nation_flag has_global_flag = nfc_chose_nation_flag }
					one_v_one_nhy_mission_select_asaf_tag = yes
					one_v_one_mission_effect_nsi = yes
				}
			}
		}
	}
}
