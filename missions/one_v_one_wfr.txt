one_v_one_mission_s4_1_wfr = {
	slot = 1
	generic = no

	one_v_one_mission_s4_nation_select_1_wfr = {
		icon = letat_cest_moi

			if = {
				limit = { OR = { has_country_flag = wfr_chose_nation_louisxiv_flag has_country_flag = wfr_chose_nation_vendome_flag has_country_flag = wfr_chose_nation_villars_flag has_country_flag = wfr_chose_nation_neufville_flag } }

			custom_trigger_tooltip = {
				tooltip = one_v_one_nation_choosing_flag_tt

				has_global_flag = wsp_nation_choosing_flag
				has_global_flag = wau_nation_choosing_flag
				has_global_flag = wgb_nation_choosing_flag

			if = {
				limit = { has_global_flag = wau_ban_louisxiv_flag }
				custom_trigger_tooltip = {
					tooltip = one_v_one_nation_banned_flag_tt
					always = no
				}
			}
		}
		effect = {
			set_country_flag = wfr_chose_nation_louisxiv_flag
			set_global_flag = wfr_chose_nation_flag
			custom_tooltip = one_v_one_chose_nation_flag_tt

				if = {
					limit = { has_global_flag = wsp_chose_nation_flag has_global_flag = wau_chose_nation_flag has_global_flag = wgb_chose_nation_flag }
					one_v_one_wfr_mission_select_louisxiv_tag = yes

						if = {
							limit = { has_country_flag = wsp_chose_nation_philip_flag }
							one_v_one_wsp_mission_select_philip_tag = yes
						}
						if = {
							limit = { has_country_flag = wsp_chose_nation_isidoro_flag }
							one_v_one_wsp_mission_select_isidoro_tag = yes
						}
						if = {
							limit = { has_country_flag = wsp_chose_nation_jorge_flag }
							one_v_one_wsp_mission_select_jorge_tag = yes
						}
						if = {
							limit = { has_country_flag = wsp_chose_nation_francisco_flag }
							one_v_one_wsp_mission_select_francisco_tag = yes
						}
					}
					WAU = {
						if = {
							limit = { has_country_flag = wau_chose_nation_charles_flag }
							one_v_one_wau_mission_select_charles_tag = yes
						}
						if = {
							limit = { has_country_flag = wau_chose_nation_leopold_flag }
							one_v_one_wau_mission_select_leopold_tag = yes
						}
						if = {
							limit = { has_country_flag = wau_chose_nation_joseph_flag }
							one_v_one_wau_mission_select_joseph_tag = yes
						}
						if = {
							limit = { has_country_flag = wau_chose_nation_eugene_flag }
							one_v_one_wau_mission_select_eugene_tag = yes
						}
					}
					WGB = {
						if = {
							limit = { has_country_flag = wgb_chose_nation_anne_flag }
							one_v_one_wgb_mission_select_anne_tag = yes
						}
						if = {
							limit = { has_country_flag = wgb_chose_nation_henry_flag }
							one_v_one_wgb_mission_select_henry_tag = yes
						}
						if = {
							limit = { has_country_flag = wgb_chose_nation_john_flag }
							one_v_one_wgb_mission_select_john_tag = yes
						}
						if = {
