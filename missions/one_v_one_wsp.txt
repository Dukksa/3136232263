one_v_one_mission_s4_1_wsp = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		NOT = { map_setup = map_setup_random }
		tag = WSP
	}
	has_country_shield = yes

	one_v_one_mission_s4_nation_select_1_wsp = {
		icon = the_spanish_succession
		required_missions = { }
		trigger = {
			if = {
				limit = { OR = { has_country_flag = wsp_chose_nation_philip_flag has_country_flag = wsp_chose_nation_isidoro_flag has_country_flag = wsp_chose_nation_jorge_flag has_country_flag = wsp_chose_nation_francisco_flag } }
				custom_trigger_tooltip = {
					tooltip = one_v_one_already_chose_nation_flag_tt
					always = no
				}
			}
			else = { tag = WSP }
			custom_trigger_tooltip = {
				tooltip = one_v_one_nation_choosing_flag_tt
				has_global_flag = wsp_nation_choosing_flag
				has_global_flag = wfr_nation_choosing_flag
				has_global_flag = wau_nation_choosing_flag
				has_global_flag = wgb_nation_choosing_flag
			}
			if = {
				limit = { has_global_flag = wgb_ban_philip_flag }
				custom_trigger_tooltip = {
					tooltip = one_v_one_nation_banned_flag_tt
					always = no
				}
			}
		}
		effect = {
			set_country_flag = wsp_chose_nation_philip_flag
			set_global_flag = wsp_chose_nation_flag
			custom_tooltip = one_v_one_chose_nation_flag_tt
			tooltip = { one_v_one_wsp_mission_select_philip_tag = yes }
			hidden_effect = {
				if = {
					limit = { has_global_flag = wfr_chose_nation_flag has_global_flag = wau_chose_nation_flag has_global_flag = wgb_chose_nation_flag }
					one_v_one_wsp_mission_select_philip_tag = yes
					WFR = {
						if = {
							limit = { has_country_flag = wfr_chose_nation_louisxiv_flag }
							one_v_one_wfr_mission_select_louisxiv_tag = yes
						}
						if = {
							limit = { has_country_flag = wfr_chose_nation_vendome_flag }
							one_v_one_wfr_mission_select_vendome_tag = yes
						}
						if = {
							limit = { has_country_flag = wfr_chose_nation_villars_flag }
							one_v_one_wfr_mission_select_villars_tag = yes
						}
						if = {
							limit = { has_country_flag = wfr_chose_nation_neufville_flag }
							one_v_one_wfr_mission_select_neufville_tag = yes
						}
					}
					WAU = {
						if = {
							limit = { has_country_flag = wau_chose_nation_charles_flag }
							one_v_one_wau_mission_select_charles_tag = yes
						}
						if = {
							limit = { has_country_flag = wau_chose_nation_leopold_flag }
							one_v_one_wau_mission_select_leopold_tag = yes
						}
						if = {
							limit = { has_country_flag = wau_chose_nation_joseph_flag }
							one_v_one_wau_mission_select_joseph_tag = yes
						}
						if = {
							limit = { has_country_flag = wau_chose_nation_eugene_flag }
							one_v_one_wau_mission_select_eugene_tag = yes
						}
					}
					WGB = {
						if = {
							limit = { has_country_flag = wgb_chose_nation_anne_flag }
							one_v_one_wgb_mission_select_anne_tag = yes
						}
						if = {
							limit = { has_country_flag = wgb_chose_nation_henry_flag }
							one_v_one_wgb_mission_select_henry_tag = yes
						}
						if = {
							limit = { has_country_flag = wgb_chose_nation_john_flag }
							one_v_one_wgb_mission_select_john_tag = yes
						}
						if = {
							limit = { has_country_flag = wgb_chose_nation_george_flag }
							one_v_one_wgb_mission_select_george_tag = yes
						}
					}
				}
			}
		}
	}
}
