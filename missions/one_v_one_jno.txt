one_v_one_mission_jap_1_jno = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		NOT = { map_setup = map_setup_random }
		OR = { tag = JNO tag = JND tag = JNK tag = JNT tag = JSI }
	}
	has_country_shield = yes

	one_v_one_mission_jap_nation_select_1_jno = {
		icon = one_v_one_mission_flag_jnd
		required_missions = { }
		trigger = {
			if = {
				limit = { OR = { has_country_flag = jno_chose_nation_jnd_flag has_country_flag = jno_chose_nation_jnk_flag has_country_flag = jno_chose_nation_jnt_flag has_country_flag = jno_chose_nation_jsi_flag } }
				custom_trigger_tooltip = {
					tooltip = one_v_one_already_chose_nation_flag_tt
					always = no
				}
			}
			else = { OR = { tag = JNO tag = JND } }
			custom_trigger_tooltip = {
				tooltip = one_v_one_nation_choosing_flag_tt
				has_global_flag = jso_nation_choosing_flag
				has_global_flag = jno_nation_choosing_flag 
			}
			if = {
				limit = { has_global_flag = jso_ban_jnd_flag }
				custom_trigger_tooltip = {
					tooltip = one_v_one_nation_banned_flag_tt
					always = no
				}
			}
		}
		effect = {
			set_country_flag = jno_chose_nation_jnd_flag
			set_global_flag = jno_chose_nation_flag
			custom_tooltip = one_v_one_chose_nation_flag_tt
			tooltip = { one_v_one_jno_mission_select_jnd_tag = yes }
			hidden_effect = {
				if = {
					limit = { has_global_flag = jso_chose_nation_flag }
					one_v_one_jno_mission_select_jnd_tag = yes
					JSO = {
						if = {
							limit = { has_country_flag = jso_chose_nation_jss_flag }
							one_v_one_jso_mission_select_jss_tag = yes
						}
						if = {
							limit = { has_country_flag = jso_chose_nation_jsm_flag }
							one_v_one_jso_mission_select_jsm_tag = yes
						}
						if = {
							limit = { has_country_flag = jso_chose_nation_jst_flag }
							one_v_one_jso_mission_select_jst_tag = yes
						}
						if = {
							limit = { has_country_flag = jso_chose_nation_jnu_flag }
							one_v_one_jno_mission_select_jnu_tag = yes
						}
					}
				}
			}
		}
	}
}
