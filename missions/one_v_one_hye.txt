one_v_one_mission_hyw_1_hye = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		NOT = { map_setup = map_setup_random }
		tag = HYE
	}
	has_country_shield = yes

	one_v_one_mission_hyw_nation_select_1_hye = {
		icon = mission_unknown_mission
		required_missions = { }
		trigger = {
			if = {
				limit = { OR = { has_country_flag = hye_chose_nation_woodstock_flag has_country_flag = hye_chose_nation_edward_flag has_country_flag = hye_chose_nation_henry_flag } }
				custom_trigger_tooltip = {
					tooltip = one_v_one_already_chose_nation_flag_tt
					always = no
				}
			}
			else = { tag = HYE }
			custom_trigger_tooltip = {
				tooltip = one_v_one_nation_choosing_flag_tt
				has_global_flag = hyf_nation_choosing_flag
				has_global_flag = hyc_nation_choosing_flag
				has_global_flag = hys_nation_choosing_flag
				has_global_flag = hye_nation_choosing_flag
				has_global_flag = hyb_nation_choosing_flag
				has_global_flag = hyp_nation_choosing_flag
			}
			if = {
				limit = { NOT = { has_global_flag = hyw_ban_chosen_flag } }
				hidden_trigger = {
					if = {
						limit = { has_global_flag = hyf_ban_woodstock_flag }
						custom_trigger_tooltip = {
							tooltip = one_v_one_nation_banned_flag_tt
							always = no
						}
					}
				}
			}
			else = {
				if = {
					limit = { has_global_flag = hyf_ban_woodstock_flag }
					custom_trigger_tooltip = {
						tooltip = one_v_one_nation_banned_flag_tt
						always = no
					}
				}
			}
		}
		effect = {
			set_country_flag = hye_chose_nation_woodstock_flag
			set_global_flag = hye_chose_nation_flag
			custom_tooltip = one_v_one_chose_nation_flag_tt
			tooltip = { one_v_one_hye_mission_select_woodstock_tag = yes }
			hidden_effect = {
				if = {
					limit = { has_global_flag = hyf_chose_nation_flag has_global_flag = hyc_chose_nation_flag has_global_flag = hys_chose_nation_flag has_global_flag = hyb_chose_nation_flag has_global_flag = hyp_chose_nation_flag }
					one_v_one_hye_mission_select_woodstock_tag = yes
					one_v_one_mission_effect_hyw = yes
				}
			}
		}
	}
}
