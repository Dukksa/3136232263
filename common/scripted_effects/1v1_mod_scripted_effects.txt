		cede_province = ROOT
		remove_core = $neutr$
		change_religion = ROOT
		change_culture = ROOT
		remove_building = fort_15th
		every_neighbor_province = { limit = { owned_by = $neutr$ } discover_country = ROOT }
		ROOT = {
			every_ally = {
				every_province = {
					limit = {
						OR = {
							AND = { owned_by = $neutr$ is_neighbor_of = ROOT }
							AND = { owned_by = ROOT has_discovered = ROOT }
						}
					}
					discover_country = PREV
				}
			}
		}
	}
	else = {
		kill_units = { who = ROOT }
		change_controller = $neutr$
	}
}

one_v_one_soldier_add = {
	while = {
		limit = { NOT = { check_variable = { which = soldiersbuilt value = 12 } } }
		if = {
			limit = {
				any_owned_province = {
					num_free_building_slots = 1
					NOT = { has_building = soldier_households }
					OR = { trade_goods = grain trade_goods = fish trade_goods = livestock trade_goods = wine }
				}
			}
			random_owned_province = {
				limit = {
					num_free_building_slots = 1
					NOT = { has_building = soldier_households }
					OR = { trade_goods = grain trade_goods = fish trade_goods = livestock trade_goods = wine }
				}
				add_building = soldier_households
				ROOT = { change_variable = { which = soldiersbuilt value = 1 } }
			}
		}
		else_if = {
			limit = {
				any_owned_province = {
					num_free_building_slots = 1
					NOT = { has_building = soldier_households }
				}
			}
			random_owned_province = {
				limit = {
					num_free_building_slots = 1
					NOT = { has_building = soldier_households }
				}
				add_building = soldier_households
				ROOT = { change_variable = { which = soldiersbuilt value = 1 } }
			}
		}
	}
}
