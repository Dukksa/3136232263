one_v_one_fort_per_area_trigger = {
	if = {
		limit = { NOT = { OR = { started_in = 1807.1.1 started_in = 1807.1.2 } } }
		if = {
			limit = {
				FROM = { tag = SPO }
			}
			NOT = {
				area_for_scope_province = {
					owner = { tag = ROOT }
					NOT = { area = beieras_area }
					NOT = { area = alentejo_area }
					OR = {
						has_building = fort_15th
						has_construction = fort_15th
						has_building = fort_16th
						has_construction = fort_16th
						has_building = fort_17th
						has_construction = fort_17th
						has_building = fort_18th
						has_construction = fort_18th
					}
				}
			}
		}
		else = {
			NOT = {
				area_for_scope_province = {
					owner = { tag = ROOT }
					OR = {
						has_building = fort_15th
						has_construction = fort_15th
						has_building = fort_16th
						has_construction = fort_16th
						has_building = fort_17th
						has_construction = fort_17th
						has_building = fort_18th
						has_construction = fort_18th
					}
				}
			}
		}
	}
}
